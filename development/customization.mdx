## Overview

The Cloro de Hidalgo Procurement System is built with customization in mind. This guide covers common customization scenarios.

## Adding New Catalogs

### 1. Create Database Table

Add to `supabase/schema.sql`:

```sql
CREATE TABLE mi_nuevo_catalogo (
  id          UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  nombre      TEXT NOT NULL,
  descripcion TEXT,
  activo      BOOLEAN NOT NULL DEFAULT TRUE,
  created_at  TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

-- Enable RLS
ALTER TABLE mi_nuevo_catalogo ENABLE ROW LEVEL SECURITY;

-- RLS Policies
CREATE POLICY "mi_nuevo_catalogo: all authenticated can select" 
ON mi_nuevo_catalogo
FOR SELECT TO authenticated USING (TRUE);

CREATE POLICY "mi_nuevo_catalogo: admin full access" 
ON mi_nuevo_catalogo
FOR ALL TO authenticated 
USING (get_my_role() = 'admin')
WITH CHECK (get_my_role() = 'admin');

-- Seed data
INSERT INTO mi_nuevo_catalogo (nombre, descripcion) VALUES
  ('Item 1', 'Descripción 1'),
  ('Item 2', 'Descripción 2');
```

Run this SQL in Supabase SQL Editor.

### 2. Add TypeScript Types

Update `types/index.ts`:

```typescript
export interface MiNuevoCatalogo {
    id: string
    nombre: string
    descripcion: string | null
    activo: boolean
    created_at: string
}

// Update Catalogos interface

    proveedores: Proveedor[]
    productos: Producto[]
    // ... existing catalogs
    mi_nuevo_catalogo: MiNuevoCatalogo[]
}
```

### 3. Add Validation Schema

Update `lib/validations/schemas.ts`:

```typescript
export const miNuevoCatalogoSchema = z.object({

    activo: z.boolean().default(true),
```

### 4. Create Server Actions

Create `lib/actions/mi-nuevo-catalogo.ts`:

```typescript
'use server'

import { createClient } from '@/lib/supabase/server'
import { getCurrentProfile } from './auth'
import { revalidatePath } from 'next/cache'

export async function createMiNuevoCatalogo(data: { nombre: string; descripcion?: string }) {
    const supabase = await createClient()
    const profile = await getCurrentProfile()

    if (profile?.rol !== 'admin') {
        return { error: 'No tienes permisos' }
    }

    const { data: result, error } = await supabase
        .from('mi_nuevo_catalogo')
        .insert(data)
        .select()
        .single()

    if (error) return { error: error.message }

    revalidatePath('/dashboard/catalogos')
    return { data: result }
}

export async function getMiNuevoCatalogo() {
    const supabase = await createClient()
    const { data, error } = await supabase
        .from('mi_nuevo_catalogo')
        .select('*')
        .order('nombre')

    if (error) return { error: error.message }
    return { data: data ?? [] }
}
```

### 5. Update getCatalogos Action

Update `lib/actions/requisiciones.ts`:

```typescript
export async function getCatalogos() {
    const supabase = await createClient()

    const [
        proveedores,

    ] = await Promise.all([

        supabase.from('productos').select('*').order('nombre'),

    return {
        proveedores: proveedores.data ?? [],
        productos: productos.data ?? [],
        // ... existing catalogs
        mi_nuevo_catalogo: miNuevoCatalogo.data ?? [],
    }
}
```

## Extending User Roles

### 1. Update Database Enum

Run in Supabase SQL Editor:

```sql
-- Add new role to enum
ALTER TYPE user_role ADD VALUE 'mi_nuevo_rol';

-- Or recreate enum (requires dropping dependent objects)
DROP TYPE user_role CASCADE;
CREATE TYPE user_role AS ENUM ('admin', 'coordinadora', 'consulta', 'mi_nuevo_rol');

-- Recreate profiles table
CREATE TABLE profiles (
  id              UUID PRIMARY KEY REFERENCES auth.users(id) ON DELETE CASCADE,
  nombre_completo TEXT,
  rol             user_role NOT NULL DEFAULT 'consulta',
  created_at      TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  updated_at      TIMESTAMPTZ NOT NULL DEFAULT NOW()
);
```

### 2. Update TypeScript Types

Update `types/index.ts`:

```typescript
export type UserRole = 
    | 'admin' 
    | 'coordinadora' 
    | 'consulta'
    | 'mi_nuevo_rol'
```

### 3. Create RLS Policies

Define permissions for new role:

```sql
-- Example: Allow mi_nuevo_rol to read requisitions
CREATE POLICY "mi_nuevo_rol can view requisiciones" ON requisiciones
  FOR SELECT TO authenticated
  USING (get_my_role() = 'mi_nuevo_rol');

-- Example: Allow mi_nuevo_rol to update status
CREATE POLICY "mi_nuevo_rol can update status" ON requisiciones
  FOR UPDATE TO authenticated
  USING (get_my_role() = 'mi_nuevo_rol')
  WITH CHECK (get_my_role() = 'mi_nuevo_rol');
```

## Custom Status Workflows

### 1. Add New Status Values

Update database:

```sql
INSERT INTO estatus (nombre, color_hex) VALUES
  ('En Validación',  '#8B5CF6'),
  ('Aprobado',       '#059669'),
  ('Rechazado',      '#DC2626');
```

### 2. Create Status Transition Rules

Define allowed transitions:

```typescript
// lib/workflows/status-transitions.ts
export const STATUS_TRANSITIONS: Record<string, string[]> = {
  'Pendiente': ['Confirmado', 'Cancelado'],
  'Confirmado': ['En Tránsito', 'Cancelado'],
  'En Tránsito': ['Recibido', 'En Revisión'],
  'En Revisión': ['Recibido', 'Rechazado'],
  'Recibido': [], // Terminal state
  'Cancelado': [], // Terminal state
  'Rechazado': ['Pendiente'], // Can retry
}

export function canTransitionTo(currentStatus: string, newStatus: string): boolean {
    return STATUS_TRANSITIONS[currentStatus]?.includes(newStatus) ?? false
}
```

### 3. Enforce in Server Actions

```typescript
import { canTransitionTo } from '@/lib/workflows/status-transitions'

export async function updateRequisicionStatus(
    id: string, 
    newStatusId: string
) {
    const supabase = await createClient()
    
    // Get current requisition
    const { data: requisicion } = await supabase
        .from('requisiciones')
        .select('*, estatus:estatus(nombre)')
        .eq('id', id)
        .single()
    
    if (!requisicion) {
        return { error: 'Requisición no encontrada' }
    }
    
    // Get new status name
    const { data: newStatus } = await supabase
        .from('estatus')
        .select('nombre')
        .eq('id', newStatusId)
        .single()
    
    if (!newStatus) {
        return { error: 'Estatus no válido' }
    }
    
    // Validate transition
    const currentStatusName = requisicion.estatus.nombre
    if (!canTransitionTo(currentStatusName, newStatus.nombre)) {
        return { 
            error: `No se puede cambiar de "${currentStatusName}" a "${newStatus.nombre}"` 
        }
    }
    
    // Perform update
    const { error } = await supabase
        .from('requisiciones')
        .update({ estatus_id: newStatusId })
        .eq('id', id)
    
    if (error) return { error: error.message }
    
    revalidatePath('/dashboard/calendar')
    return { success: true }
}
```

### 4. Update UI with Allowed Transitions


## Adding Email Notifications

### 1. Install Email Service

```bash
npm install @sendgrid/mail
# or
npm install nodemailer
```

### 2. Create Email Action

```typescript
// lib/actions/notifications.ts
'use server'

import sgMail from '@sendgrid/mail'

sgMail.setApiKey(process.env.SENDGRID_API_KEY!)

export async function sendStatusChangeEmail(
    recipientEmail: string,
    requisicionId: string,
    oldStatus: string,
    newStatus: string
) {
    const msg = {
        to: recipientEmail,
        from: 'notificaciones@clorohidalgo.com',
        subject: 'Cambio de estatus en requisición',
        html: `
            <h1>Actualización de Requisición</h1>
            <p>La requisición ${requisicionId} cambió de:</p>
            <p><strong>${oldStatus}</strong> → <strong>${newStatus}</strong></p>
        `,
    }
    
    try {
        await sgMail.send(msg)
        return { success: true }
    } catch (error) {
        console.error('Error sending email:', error)
        return { error: 'No se pudo enviar el correo' }
    }
}
```

## Best Practices

<Steps>

### Keep Changes Modular

Organize customizations in separate files:
- New catalogs → `lib/actions/[catalog-name].ts`
- Custom workflows → `lib/workflows/`
- Email templates → `lib/templates/`

### Document Custom Code

Add comments for custom logic:

```typescript
// CUSTOM: Added by [Your Name] on [Date]
// Purpose: Allow special users to bypass approval
export async function customApprovalLogic() {
    // ...
}
```

### Test Thoroughly

Before deploying customizations:
1. Test in development
2. Verify RLS policies
3. Check all user roles
4. Test error scenarios
5. Deploy to preview environment first

### Version Control

Use Git branches for customizations:

```bash
git checkout -b feature/custom-catalog
# Make changes
git add .
git commit -m "Add custom catalog for [Feature]"
git push origin feature/custom-catalog
```

</Steps>
